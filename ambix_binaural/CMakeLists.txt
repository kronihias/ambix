SET ( SPECIFIC_SOURE_DIR ${SRC_DIR}/ambix_binaural )

SET ( WITH_MyMeter TRUE )
SET ( WITH_MyMeterDsp TRUE )

IF(NOT APPLE)
	SET ( WITH_FFTW3 TRUE )
ENDIF(NOT APPLE)

ADD_DEFINITIONS (-DBINAURAL_DECODER=1)

# VST2 fixed-order
IF(BUILD_VST)
	SET(UNIVERSAL_BUILD FALSE)
	SET( PLUGIN_CODE ABi${AMBI_ORDER})
	INCLUDE(${SRC_DIR}/Subproject.cmake)
ENDIF()

# VST3/AU universal
IF(BUILD_VST3 OR BUILD_AU)
	SET(UNIVERSAL_BUILD TRUE)
	SET( PLUGIN_CODE ABiU)
	INCLUDE(${SRC_DIR}/Subproject.cmake)
ENDIF()
